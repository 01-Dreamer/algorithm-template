#include<iostream>
#include<cstring>

using namespace std;
const int N=200003,INF=0x3f3f3f3f;
int h[N];
int n;

int find(int x)
{
    int k=(x%N+N)%N;
    
    while(h[k]!=INF&&h[k]!=x)
    {
        k++;
        if(k==N) k=0;
    }
    return k;
}


int main()
{
    
    memset(h,0x3f,sizeof h);
