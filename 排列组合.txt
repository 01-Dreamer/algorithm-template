typedef long long ll;
ll C(ll n,ll m,ll mod)
{
    if(m>n) return 0;
    if(2*m>n) m=n-m;
    ll t=m,A=1,B=1;
    while(t--)
    {
        A=A*n--%mod;
        B=B*m--%mod;
    }
    return A* fast_pow(B,mod-2,mod)%mod;
}

ll A(int n,int m,int mod)
{
    ll result=1;
    while(m--)
        result=result*n--%mod;
    return result;
}
